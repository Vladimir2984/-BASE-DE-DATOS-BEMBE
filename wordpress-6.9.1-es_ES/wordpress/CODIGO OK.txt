// ==================== STARRY BACKGROUND ====================
function initStarryBg() {
const canvas = document.getElementById('starCanvas');
if (!canvas) return;
const ctx = canvas.getContext('2d');
let frame = 0;
const stars = [];
const colors = ['255,255,255','200,220,255','180,200,255','255,240,210','180,230,255'];
function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
function init() {
resize();
stars.length = 0;
for (let i = 0; i < 350; i++) {
stars.push({
x: Math.random()*canvas.width, y: Math.random()*canvas.height,
r: Math.random()*1.8+0.2, baseOp: Math.random()*0.5+0.2,
tSpeed: Math.random()*0.025+0.005, tOff: Math.random()*Math.PI*2,
color: colors[Math.floor(Math.random()*colors.length)],
});
}
}
init();
window.addEventListener('resize', resize);
function drawNeb(cx,cy,rad,col) {
const g = ctx.createRadialGradient(cx,cy,0,cx,cy,rad);
g.addColorStop(0,col); g.addColorStop(1,'transparent');
ctx.fillStyle = g; ctx.fillRect(0,0,canvas.width,canvas.height);
}
function animate() {
const w=canvas.width, h=canvas.height;
const bg = ctx.createLinearGradient(0,0,w*0.4,h);
bg.addColorStop(0,'#020a18'); bg.addColorStop(0.25,'#071a35');
bg.addColorStop(0.5,'#0c2248'); bg.addColorStop(0.75,'#091d3a'); bg.addColorStop(1,'#040e1f');
ctx.fillStyle = bg; ctx.fillRect(0,0,w,h);
drawNeb(w*0.15,h*0.25,w*0.45,'rgba(30,64,175,0.035)');
drawNeb(w*0.8,h*0.65,w*0.35,'rgba(99,40,180,0.025)');
drawNeb(w*0.5,h*0.8,w*0.3,'rgba(6,182,212,0.02)');
drawNeb(w*0.7,h*0.2,w*0.25,'rgba(251,191,36,0.012)');
stars.forEach(s => {
const op = Math.max(0.05,Math.min(1,s.baseOp+Math.sin(frame*s.tSpeed+s.tOff)*0.35));
ctx.beginPath(); ctx.arc(s.x,s.y,s.r,0,Math.PI*2);
ctx.fillStyle = `rgba(${s.color},${op})`; ctx.fill();
if (s.r > 1.2) {
ctx.beginPath(); ctx.arc(s.x,s.y,s.r*3.5,0,Math.PI*2);
ctx.fillStyle = `rgba(${s.color},${op*0.06})`; ctx.fill();
}
});
frame++;
requestAnimationFrame(animate);
}
requestAnimationFrame(animate);
}

function renderStudentList() {
const container = document.getElementById('studentListContainer');
if (!container) return;
const q = (document.getElementById('searchInput')?.value || '').toLowerCase();
const filtered = appData.students.filter(s => {
const full = `${s.surname} ${s.name}`.toLowerCase();
const full2 = `${s.name} ${s.surname}`.toLowerCase();
return full.includes(q) || full2.includes(q);
});
if (filtered.length === 0) {
container.innerHTML = '<p style="text-align:center;color:rgba(147,197,253,0.3);font-size:0.875rem;padding:16px;">No hay alumnos</p>';
return;
}
const now = new Date();
container.innerHTML = filtered.map(s => {
const hasSub = !!(s.start && s.end);
let diffDays = null;
if (hasSub) diffDays = Math.ceil((new Date(s.end).getTime()-now.getTime())/86400000);
const expColor = diffDays !== null && diffDays < 5 ? '#f43f5e' : '#10b981';
return `
<div class="student-card" style="background:rgba(30,58,138,0.25);border-radius:8px;padding:12px;margin-bottom:8px;cursor:pointer;" onclick="fillQuickForm('${s.id}', '${s.surname}', '${s.name}', '${s.phone || ''}', '${s.birthDate || ''}')">
<h4 style="font-weight:700;color:white;font-size:0.875rem;">${s.surname} ${s.name}</h4>
<p style="font-size:11px;color:rgba(147,197,253,0.5);">${s.phone||'Sin teléfono'}</p>
${hasSub ? `<p style="font-size:11px;"><span style="color:rgba(147,197,253,0.5);">${s.hours}h</span> | <span style="color:${expColor};">Vence: ${s.end}</span></p>` : '<p style="font-size:11px;color:rgba(147,197,253,0.3);">Sin abono activo</p>'}
</div>`;
}).join('');
}
function renderDashboard() {
const now = new Date();
let income = 0, hrs = 0, expiring = 0;
appData.students.forEach(s => {
income += (s.cost || 0);
const h = parseInt(s.hours);
if (!isNaN(h) && h < 99) hrs += h;
if (s.end) {
const diff = Math.ceil((new Date(s.end).getTime()-now.getTime())/86400000);
if (diff <= 5 && diff >= 0) expiring++;
}
});
const classes = Math.floor(hrs * 4);
const statsEl = document.getElementById('dashStats');
statsEl.innerHTML = [
{label:'Total',val:appData.students.length,color:'#fbbf24'},
{label:'Profesores',val:appData.teachers.length,color:'#22d3ee'},
{label:'Horas Sem.',val:hrs,color:'#60a5fa'},
{label:'Vence pronto',val:expiring,color:'#fb7185'},
].map(s => `
<div class="stat-card">
<div style="font-size:10px;color:rgba(147,197,253,0.5);text-transform:uppercase;letter-spacing:1px;">${s.label}</div>
<div style="font-size:1.5rem;font-weight:800;color:${s.color};">${s.val}</div>
</div>
`).join('');
const econEl = document.getElementById('dashEcon');
econEl.innerHTML = `
<div style="text-align:center;color:rgba(251,191,36,0.8);font-size:0.75rem;font-weight:700;border-bottom:1px solid rgba(255,255,255,0.05);padding-bottom:8px;margin-bottom:12px;">📈 Resumen Económico</div>

<span style="color:#22d3ee;font-weight:700;">${classes}</span>
</div>
`;
}

function deleteSelectedStudent() {
  if (!selectedStudentId) {
    showToast("❌ No se ha seleccionado ningún alumno");
    return;
  }
  if (confirm("¿Estás seguro de eliminar este alumno?")) {
    appData.students = appData.students.filter(student => student.id != selectedStudentId);
    saveData();
    document.getElementById('qfSurname').value = '';
    document.getElementById('qfName').value = '';
    document.getElementById('qfPhone').value = '';
    document.getElementById('qfBirth').value = '';
    document.getElementById('qfRegister').value = todayStr();
    selectedStudentId = null;
    document.getElementById('qfDeleteBtn').disabled = true;
    renderStudentList();
    renderDashboard();
    showToast("✅ Alumno eliminado correctamente");
  }
}

document.getElementById('qfBirth').value = '';
document.getElementById('qfRegister').value = todayStr();

function saveSubscription() {
const surname = document.getElementById('sfSurname').value.trim();
const name = document.getElementById('sfName').value.trim();
const start = document.getElementById('sfStart').value;
const end = document.getElementById('sfEnd').value;
const price = document.getElementById('sfPrice').value;
const hours = document.getElementById('sfHours').value;
if (!surname || !name) { alert('Apellido y nombre requeridos'); return; }
if (!hours || !price) { alert('Complete horas y precio'); return; }
const priceNum = parseFloat(price) || 0;
// Find existing student
const idx = appData.students.findIndex(s =>
s.surname.toLowerCase() === surname.toLowerCase() &&
s.name.toLowerCase() === name.toLowerCase()
);
if (idx !== -1) {
// Update existing student
appData.students[idx].hours = hours;
appData.students[idx].cost = priceNum;
appData.students[idx].start = start;
appData.students[idx].end = end;
} else {
// Create new student with subscription
appData.students.push({
id: Date.now() + Math.random(),
name, surname, phone: '',
birthDate: null, registerDate: todayStr(),
hours, cost: priceNum, start, end
});
}
// Add to history
appData.history.push({
id: Date.now() + Math.random(),
studentName: `${surname} ${name}`,
type: `${hours} Horas/Clases`,
price: priceNum,
datePurchased: new Date().toLocaleDateString('es-ES'),
period: `${start} → ${end}`
});
saveData();
renderStudentList();
renderDashboard();
closeModal();
showToast(`✅ Abono para "${surname} ${name}" guardado correctamente.`);
}

function addPaymentToDebt(id) {
const input = document.getElementById('debtPay_' + id);
const amount = parseFloat(input?.value) || 0;
if (amount <= 0) { alert('Ingrese un monto válido'); return; }
const debt = appData.debtRecords.find(d => d.id === id);
if (!debt) return;
debt.paid = (debt.paid || 0) + amount;
if (debt.paid > debt.total) debt.paid = debt.total;
saveData();
openModal('debtors');
showToast(`✅ Pago de ${amount}₽ registrado.`);
}

function renderManageHistory() {
const container = document.getElementById('manageHistoryBody');
if (!container || !manageId) return;
const s = appData.students.find(st => st.id === manageId);
if (!s) { container.innerHTML = '<tr><td colspan="4" style="text-align:center;color:rgba(147,197,253,0.3);padding:16px;">Sin historial.</td></tr>'; return; }
const hist = appData.history.filter(h => {
const hn = h.studentName.toLowerCase().replace(/\s+/g,'');
const sn1 = `${s.surname}${s.name}`.toLowerCase().replace(/\s+/g,'');
const sn2 = `${s.name}${s.surname}`.toLowerCase().replace(/\s+/g,'');
return hn.includes(sn1) || hn.includes(sn2) || sn1.includes(hn) || sn2.includes(hn);
}).reverse();
if (hist.length === 0) {
container.innerHTML = '<tr><td colspan="4" style="text-align:center;color:rgba(147,197,253,0.3);padding:16px;font-size:0.875rem;">Sin historial.</td></tr>';
return;
}
container.innerHTML = hist.map(h => `
<tr style="border-bottom:1px solid rgba(255,255,255,0.05);">
<td style="padding:8px 12px;color:rgba(251,191,36,0.8);font-size:0.75rem;">${h.datePurchased}</td>
<td style="padding:8px 12px;color:rgba(191,219,254,0.7);font-size:0.75rem;">${h.type}</td>
<td style="padding:8px 12px;color:rgba(191,219,254,0.5);font-size:11px;">${h.period}</td>
<td style="padding:8px 12px;color:#34d399;font-weight:700;font-size:0.75rem;">${h.price}₽</td>
</tr>
`).join('');
}
function saveManageDetails() {
if (!manageId) return;
const idx = appData.students.findIndex(s => s.id === manageId);
if (idx === -1) return;
const old = appData.students[idx];
const newData = {
surname: document.getElementById('mfSurname').value.trim(),
name: document.getElementById('mfName').value.trim(),
phone: document.getElementById('mfPhone').value.trim(),
cost: parseFloat(document.getElementById('mfCost').value) || 0,
hours: document.getElementById('mfHours').value.trim(),
start: document.getElementById('mfStart').value,
end: document.getElementById('mfEnd').value,
};
appData.students[idx] = { ...old, ...newData };
// Add history if subscription changed
if (old.hours !== newData.hours || old.cost !== newData.cost || old.start !== newData.start || old.end !== newData.end) {
if (newData.hours && newData.cost) {
appData.history.push({
id: Date.now() + Math.random(),
studentName: `${newData.surname} ${newData.name}`,
type: `${newData.hours} Horas/Clases`,
price: newData.cost,
datePurchased: new Date().toLocaleDateString('es-ES'),
period: `${newData.start} → ${newData.end}`
});
}
}
saveData();
renderStudentList();
renderDashboard();
renderManageHistory();
showToast('✅ Datos actualizados.');
}

prices: imported.prices || JSON.parse(JSON.stringify(DEFAULT_PRICES)),
};
saveData();
renderAll();

}

}

// ==================== HELPER FUNCTIONS ====================
function autoFillSubName() {
const surnameInput = document.getElementById('sfSurname');
if (!surnameInput) return;
const val = surnameInput.value.trim();
// Try to find matching student
const s = appData.students.find(st => {
const full = `${st.surname} ${st.name}`;
return full === val || st.surname.toLowerCase() === val.toLowerCase();
});
if (s) {
const nameInput = document.getElementById('sfName');
if (nameInput && !nameInput.value) nameInput.value = s.name;
// If full name matched from datalist
if (`${s.surname} ${s.name}` === val) {
surnameInput.value = s.surname;
if (nameInput) nameInput.value = s.name;
}
}
}
function onPriceSelect() {
const sel = document.getElementById('sfPrice');
const hoursInput = document.getElementById('sfHours');
if (!sel || !hoursInput) return;
const opt = sel.options[sel.selectedIndex];
if (opt && opt.dataset.hours) {
hoursInput.value = opt.dataset.hours;
} else {
hoursInput.value = '';
}
}
// ==================== VALIDATION FUNCTIONS ====================
function validateRequired(value) {
  return value.trim().length > 0;
}

function validatePhone(phone) {
  const regex = /^[+\d\s\-()]{6,20}$/;
  return regex.test(phone.trim());
}

function validateEmail(email) {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email.trim());
}

function validateNumber(value) {
  return !isNaN(parseFloat(value)) && isFinite(value);
}

if (!grid) return;
const w = window.innerWidth;
if (w >= 1280) grid.style.gridTemplateColumns = 'repeat(4, 1fr)';
else if (w >= 768) grid.style.gridTemplateColumns = 'repeat(2, 1fr)';
else grid.style.gridTemplateColumns = '1fr';
}

function changeLanguage(lang) {
  currentLanguage = lang;
  renderAll();
  const langName = {es: 'ESPAÑOL', ru: 'РУССКИЙ', en: 'ENGLISH'}[lang] || lang.toUpperCase();
  showToast(`${translations[currentLanguage].languageChanged} ${langName}`);
  
  // Save language preference
  try {
    localStorage.setItem('bembe_language', lang);
  } catch(e) { console.error('Error saving language:', e); }
}

// ==================== INIT ====================
function initApp() {
  loadData();
  
  // Load saved language preference
  try {
    const savedLang = localStorage.getItem('bembe_language');
    if (savedLang && translations[savedLang]) {
      currentLanguage = savedLang;
    }
  } catch(e) { console.error('Error loading language:', e); }
  
  renderAll();
  updateGridLayout();
  window.addEventListener('resize', updateGridLayout);
}
// Start
